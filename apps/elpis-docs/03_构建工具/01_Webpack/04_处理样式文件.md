# 处理样式文件

> webpack.config.js

```js
// npm i css-loader stylus-loader sass-loader less-loader mini-css-extract-plugin css-minimizer-webpack-plugin -D 
const CssMinimizerPlugin = require("css-minimizer-webpack-plugin");
const MiniCssExtractPlugin = require("mini-css-extract-plugin");

module.exports = {
  // ...
  module: {
    rules: [
      {
        oneOf: [
          {
            test: /\.css$/,
            use: [MiniCssExtractPlugin.loader, "css-loader"],
          },
          {
            test: /\.less$/,
            use: [MiniCssExtractPlugin.loader, "css-loader", "less-loader"],
          },
          {
            test: /\.s[ac]ss$/i,
            use: [
              MiniCssExtractPlugin.loader, // style-loader 改为 MiniCssExtractPlugin.loader
              "css-loader", // 将 CSS 转化成 CommonJS 模块
              "sass-loader", // 将 Sass 编译成 CSS
            ],
          },
          {
            test: /\.styl$/,
            use: [MiniCssExtractPlugin.loader, "css-loader", "stylus-loader"],
          },
        ],
      },
    ],
  }, 
  optimization: {
    minimizer: [
      // 在 webpack@5 中，你可以使用 `...` 语法来扩展现有的 minimizer（即 `terser-webpack-plugin`），将下一行取消注释
      // `...`,
      new CssMinimizerPlugin(), // css代码压缩 （压缩为一行）
    ],
  },
};
```

## css兼容性处理

- babel处理js的兼容性问题
- postcss-loader postcss-preset-env 处理 css兼容性问题

```js
// npm i postcss-loader postcss postcss-preset-env -D
exports.modules = {
module: {
  rules: [
    test: /\.css$/,
    use: [
      'style-loader',
      'css-loader',
      // 在 css-loader 和 sass-loader之间使用
      {
        loader: 'postcss-loader',
        options: {
          postcssOptions: {
            plugins: [
              'postcss-preset-env',
            ]
          }
        }
      },
      'sass-loader'
    ]
  ],
}
}
// package.json
module.exports = {
  browserslist: [
    'ie >= 8', // 兼容性做到 ie大于等于8 都要支持。 书写display：flex 会编译为 display：-ms-flexbox;
  ],

  // 常用 意思是 兼容大部分浏览器
  browserslist: [
    'last 2 version',
    '> 1%',
    'not dead',
  ]
}
```
 
## 把css提取问单独文件

```js
// npm i mini-css-extract-plugin -D

// css两种引入方式  1 style标签内部写 2 单独css文件 link标签引入

// style标签内部写 网络慢有白屏现象


const MiniCssExtractPlugin = require('mini-css-extract-plugin');



exports.modules = {
module: {
  rules: [
    test: /\.css$/,
    use: [
      MiniCssExtractPlugin.loader, // 把'style-loader' 改为 MiniCssExtractPlugin.loader
      'css-loader',
    ]
  ],
  plugins: [
    new MiniCssExtractPlugin({
      filename: "css/[name]_[contenthash:8].css",
    }),
  ]
}
}



// 形成单独的css文件是MiniCssExtractPlugin做的
// 用link标签引入 这件事是 html模版插件做的

exports.modules = {
module: {
  rules: [
    test: /\.css$/,
    use: [
      'style-loader', // 把css 放到 style标签内部 而不是单独css文件
      'css-loader',
    ]
  ],
}
}

```
