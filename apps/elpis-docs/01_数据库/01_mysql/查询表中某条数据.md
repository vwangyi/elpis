## select 查询  (最重点)
```sql
所有查询 都用select
简单查询: 一张表查询
复杂查询: 多张表查询 关联查询 排序 分页
查询是使用频率最高的


-- 查询数据
-- 指定条件查询
select * from `表名` where 条件;  -- 查询表中所有字段
select `字段1`,`字段2` from `表名` where 条件;  -- 查询表中指定字段
-- 起别名 可以给字段或表 起别名
select `字段1` as `别名1`,`字段2` as `别名2` from `表名` as `新表名`;
-- 函数 concat(a,b) 拼接字符串


-- 把根据字段查询的结果 都加上了 `name:` 前缀 字符串
select concat(`name: `,字段名) as `新名字` from `student`;


-- 查到重复数据 需要去重
select * from `student`;
select `studentNo` from `student`;  -- 发现重复数据
select distinct `studentNo` from `student`; -- 去重
select distinct `字段名` from `表名` where 条件;  -- distinct 去重

-- 数据库的列 (表达式)
select version();  -- 查看系统版本(函数)
select 100*3-1 as `计算结果`;  -- 用来计算 (表达式)
select @@auto_increment_increment;  -- 查看自增步长 (变量)

-- 把查到的考试成绩 + 1 分
select `studentNo`, `score`+1 as `新成绩` from `student`;


-- 数据库中的表达式 文本值 列 null 函数 计算表达式 系统变量...
select 表达式 from `表名`;
```

## 联表查询  很重要 重难点
> 联表查询 join   
查询参加了考试的同学(学号 姓名 科目编号 分数)
>

```sql
select * from `student`;
select * from `result`;
```

```sql
/*
   思路
   1.分析需求 分析字段来自哪些表 (多张表用连接查询)
   2.确定使用哪种连接查询？ (7种) （本质3种 扩展7种）
      确定交叉点 (这两个表中哪个数据是相同的)
      判断的条件: 学生表中的 studentNo = 成绩表中的 studentNo
*/

-- inner join 内连接
select s.studentNo, studentName, SubjectNo, StudentResult   -- 因为两张表都有studentNo字段 所以起别名 用别名. 来指定是哪个表的字段
from student as s    -- 给student表 起别名 s
inner join result as r  -- 给result表 起别名 r
where s.studentNo = r.studentNo;   -- where 可以 改为 on 两者效果一样


-- right join  右连接   
select s.studentNo, studentName, SubjectNo, StudentResult   -- 因为两张表都有studentNo字段 所以起别名 用别名. 来指定是哪个表的字段
from student as s right join result as r    -- 表名 as 别名    这里的as 可以省略
on s.studentNo = r.studentNo;   -- where 可以 改为 on 两者效果一样


-- left join  左连接
select s.studentNo, studentName, SubjectNo, StudentResult   -- 因为两张表都有studentNo字段 所以起别名 用别名. 来指定是哪个表的字段
from student as s    -- 给student表 起别名 s    这里的as 可以省略
left join result r  -- 给result表 起别名 r   这里的as 可以省略
on s.studentNo = r.studentNo;   -- where 可以 改为 on 两者效果一样


操作                  描述 
inner join  内连接
right join  右连接
left join   左连接

```

## 3张表连接
```sql
-- 查询参加考试的同学的信息: （学号, 学生姓名, 科目名, 分数）
-- 注意 这四个字段 来自三张表
-- 1 分析需求 分析查询字段来自哪些表 student result subject (连接查询)
-- 2 确定使用 哪些连接 查询  inner join left join  right join 7种本质3种
-- 3 确定交叉点 (这两个表中 哪个数据是相同的) on
-- 4 判断的条件 : 学生表中 的studentNo = 成绩表 studentNo 

select s.studentNo, studentName, subjectName, studentResult
from student s right join result r   -- 左表是student 右表是result 以右表为基准
on r.studentNo = s.studentNo     -- 已经查出这两张表了
inner join `subject` sub   -- 和第三张表连接 左表是result 右表是subject  
on r.subjectNo = sub.subjectNo;  -- 已经查出这三张表了

-- join on 可以写多个
-- where 只能写一个 多个条件用 and 连接

-- 我要查询哪些数据  select 
-- 从哪几个表中 查 from 表 xxx join 连接的表  on 交叉条件
-- 假设存在多张表查询 慢慢来 先查询两张表 查出结果 再连接第三张表 查询 慢慢增加
```

## 排序
```sql
-- 排序可以前端做 也可以后端做
-- 排序: 升序 asc 降序 desc

-- order by 通过哪个字段排序 怎么排
-- 查询的结果 根据 成绩 降序 排序

select s.studentNo, studentName, subjectName, studentResult
from student s inner join result r
on r.studentNo = s.studentNo
inner join `subject` sub
on r.subjectNo = sub.subjectNo
where studentName = '数据库'
order by studentResult desc;  -- 通过studentResult 字段 降序排列
```

## 分页
```sql
-- 1 如果数据有 100万 条  我们不可能一次性把数据全部查出来 这时就用到 分页 
-- 为什么要用分页 缓解数据库压力 给人的体验更好   ， （一般图片用 瀑布流)

-- 分页 需求 一共100万条 每页只显示5条
-- limit 语法    limit 从第几条开始查, 查几条 (最小值是0开始)

select s.studentNo, studentName, subjectName, studentResult
from student s inner join result r
on r.studentNo = s.studentNo
inner join `subject` sub
on r.subjectNo = sub.subjectNo
where studentName = '数据库'
order by studentResult desc
limit 0,5;  -- 从第0条开始 取5条数据



-- 前端应用
-- 第一页  limit 0,5    (1-1)*5
-- 第二页  limit 5,5    (2-1)*5
-- 第三页  limit 10,5   (3-1)*5
-- 第n页                (n-1) * pageSize, pageSize

-- 【pageSize: 页面大小】
-- 【(n-1) * pageSize: 从第几条开始查】
-- 【pageSize: 查几条】
-- 【n : 当前页码】
-- 【数据总数 / 页面大小 = 总页数】



```

## select 查询完整语法
```sql
select [all | distinct] -- distinct 去重
{ * | table.* | [table.field1[as alias1][,table.field2[as alias2]][,...]]}
from 表名 [as 别名]
    [inner | left | right] join 表名 [as 别名] [on 条件]  -- 连接查询
    [where 条件] -- 指定结果需要满足的条件
    [group by 字段] -- 指定结果需满足的条件
    [having 条件] -- 过滤分组的记录 必须满足的次要条件
    [order by 字段 [asc | desc]] -- 指定查询记录按一个或多个条件排序
    [limit m,n] -- 指定查询记录的范围 从哪条 到 哪条

-- [] 表示可选  {} 表示必选
```



## 子查询
```sql
-- 子查询: 查询结果作为另一个查询的条件
-- 1. 查询 数据库结构-1 的所有考试结果 (学号 科目编号 成绩) 降序排列

-- 方式1  使用连接查询
select `studentNo`, r.`subjectNo`, `studentResult`
from result r inner join `subject` sub
on r.`subjectNo` = sub.`subjectNo`
where sub.`subjectName` = '数据库结构-1'
order by `studentResult` desc;


-- 方式2 使用子查询 (先里后外)

select `studentNo`, `subjectNo`, `studentResult`
from result
where `subjectNo` = (
  -- 先用子查询 查询出 subject 的subjectNo字段   等于  result的subjectNo
  select `subjectNo` from `subject` 
  where `subjectName` = '数据库结构-1'
)

-- 如果查询字段 多个表都有这个字段 就需要as给表取别名 通过别名指定字段了
 


-- 查询课程为 高等数学-2 且分数不小于 80 的同学的学号和姓名
-- 方式 1  使用的是连接查询 (代码多 好理解)
select s.`studentNo`, `studentName`
from student s inner join result r on s.`studentNo` = r.`studentNo`
inner join `subject` sub on r.`subjectNo` = sub.`subjectNo`
where sub.`subjectName` = '高等数学-2' and r.`studentResult` >= 80;

-- 方式 2  连接查询和子查询结合
select distinct s.`studentNo`, `studentName`
from student s inner join result r on s.`studentNo` = r.`studentNo`
where `studentResult` >= 80 and `subjectNo` = (
  select `subjectNo` from `subject` where `subjectName` = '高等数学-2'
);

-- 方式 3  子查询 和 嵌套查询 (先里后外) （代码最少 难理解）
select studentNo, studentName from student where studentNo in (
  select `studentNo` from result where `studentResult` >= 80 and `subjectNo` = (
    select `subjectNo` from `subject` where `subjectName` = '高等数学-2'
  )
);
```

/*
    什么是MD5？
    主要是 增强算法复杂度和不可逆性 
    MD5 不可逆  具体的值 的MD5是一样的
    MD5 破解网站的原理 背后有一个字典 MD5加密后的值，加密的前值
*/

-- 1. MD5 加密
create table `testmd5`(
  `id` int(4) not null,
  `name` varchar(20) not null,
  `pwd` varchar(50) not null,
  primary key(`id`)
)engine=innodb default varchar=utf8

-- 明文密码
insert into testmd5 values(1, 'zhangsan','123456'),(2, 'lisi','123456'),(3, 'wangwu','123456')

-- 加密
update testmd5 set pwd=MD5(pwd) where id = 1  -- 加密 id = 1 这一行

-- 如何比对密码？ 把用户传递的密码 进行md5 加密 然后 密文和密文比对
