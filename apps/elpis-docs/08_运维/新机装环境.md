## 安装 nvm(nodejs)
### <font style="color:rgba(0, 0, 0, 0.9);">Ubuntu</font>
```shell
# 首先确保系统已更新
sudo apt update && sudo apt upgrade -y

# 安装 curl（如果还没有）
sudo apt install -y curl

# 下载并运行 NVM 安装脚本
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.7/install.sh | bash

source ~/.bashrc
```

### CentOS
```shell

curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.5/install.sh | bash

source ~/.bashrc

nvm --version

nvm i v20.19.5


```

### mac
+ 卸载已安装的node和npm 命令行输入下面命令
    - sudo rm -rf /usr/local/{bin/{node,npm},lib/node_modules/npm,lib/node,share/man/*/node.*}
+ 通过 Homebrew 安装 nvm 命令行输入下面命令
    - brew install nvm
    - 下载 nvm 成功 却不能使用 nvm 命令 说明没有配置 nvm 的环境变量
+ 配置 nvm 环境变量
    - mac电脑的用户根目录下 按下 shift + command + . 查看隐藏文件   
    - 配置 nvm 环境变量 全部复制粘贴执行 

```plain
echo 'export NVM_DIR="$HOME/.nvm"' >> ~/.zshrc
echo '[ -s "/opt/homebrew/opt/nvm/nvm.sh" ] && \. "/opt/homebrew/opt/nvm/nvm.sh"' >> ~/.zshrc
echo '[ -s "/opt/homebrew/opt/nvm/etc/bash_completion.d/nvm" ] && \. "/opt/homebrew/opt/nvm/etc/bash_completion.d/nvm"' >> ~/.zshrc
```

    - 刷新配置
        * source ~/.zshrc 
    - 输入 nvm 测试 
+ nvm 安装 node
    - nvm install 18.19.1
    - <font style="color:rgb(77, 77, 77);">nvm alias default 18.19.1</font>
+ npm 全局安装 pnpm
    - npm i pnpm -g 

## 安装mysql8.0.19 
### Centos
```shell
# 查看 CentOS系统版本
cat /etc/redhat-release

# 下载 MySQL Yum 仓库安装包
sudo yum install -y https://dev.mysql.com/get/mysql80-community-release-el7-11.noarch.rpm
# 如果是 CentOS 8，使用：
sudo dnf install -y https://dev.mysql.com/get/mysql80-community-release-el8-6.noarch.rpm

# 安装 MySQL 服务器
sudo yum install -y mysql-community-server

# 或者分别安装客户端和服务器
sudo yum install -y mysql-community-client mysql-community-server

# 启动服务
sudo systemctl start mysqld

# 设置开机自启
sudo systemctl enable mysqld

# 检查服务状态
sudo systemctl status mysqld
# 查看临时密码
sudo grep 'temporary password' /var/log/mysqld.log
 

# 使用 root 登录 登录 MySQL (-p后面不建议跟密码 推荐直接回车手动输入密码）
mysql -u root -p  
```

### mac
+ 访问 [https://downloads.mysql.com/archives/community/](https://downloads.mysql.com/archives/community/)  选择8.0.19 安装 mysql 
+ 配置 mysql 环境变量

```shell
vim ~/.zshrc # 用 vim 编辑器 打开 ~/.zshrc 配置文件 

# 按下 i 进入编辑模式
# 按下 esc 退出编辑模式
# 退出编辑模式后 输入 :wq 按下回车 是保存 
# 关闭命令行 重新打开 重启命令行
      
PATH=$PATH:/usr/local/mysql/bin # 追加一行
# mac 会自动启动mysql 不用手动启动 
mysql --verion # 这是mysql命令
mysql -u root -p密码 # -p后面紧跟密码 不要有空格    登录mysql后就可以使用sql命令了 

show databases; # 查看所有数据库  这是 sql命令
```



## 安装 nginx
### Centos 
```shell
yum update -y # 更新系统包

# 安装 Nginx
yum install nginx -y 

# 启动并设置开机自启
systemctl start nginx
systemctl enable nginx

# 检查 Nginx 状态
systemctl status nginx

# 检查 Nginx 版本
nginx -v

在浏览器访问 http://你的服务器IP，应该能看到 Nginx 欢迎页面。
```

### mac 
+ 先安装 brew ，安装了 brew 就不用安装了
+ [https://www.jianshu.com/p/c3294887c6b6](https://www.jianshu.com/p/c3294887c6b6) 

```shell
# 通过 brew 为本机安装 nginx 
brew install nginx
# 启动 nginx 
brew services start nginx

# 成功后 浏览器输入 http://localhost:8080 就可以直接访问nginx静态资源服务器了
# nginx默认安装到 /usr/local/ 目录下 

# 重启 nginx 
brew services restart nginx

# 查看nginx目录 
which nginx


下面是常见的 nginx命令

sudo nginx -s stop # 停止nginx服务 停止后就访问不到了 localhost:8080
sudo nginx # 启动nginx服务 可以访问 localhost:8080
sudo nginx -s reload # 重启nginx服务  


# 以上步骤 可以在本地mac搭建 
# 在云服务器上也是类似方法搭建 来部署前端项目等
```
