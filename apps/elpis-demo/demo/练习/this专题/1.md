#### 案例1

```js
function fn() {
  console.log('fn:', this) // window
  var f1 = () => {
    console.log('f1:', this) // window
  }
  f1()
}

var obj = {
  f2() {
    console.log(this) // obj
    fn()
  },
  f3: () => {
    console.log(this) // window
  },
}

obj.f2()
obj.f3()
```

#### 案例2

```js
function fn() {
  console.log('fn:', this)
  var f1 = () => {
    console.log('f1:', this)
  }
  f1()
}

var obj = {
  f2() {
    fn.call(this)
  },
}

obj.f2()
```

#### 案例3

```js
const a = {
  fn() {
    // b
    console.log('fn:', this) // b
    var f1 = () => {
      console.log('f1:', this) // b
    }
    f1()
  },
}

const b = {}

var obj = {
  f2() {
    // this: obj
    // 就想让fn的this是obj
    a.fn()
    // fn运行 并且换this
    a.fn.call(b)
  },
}

obj.f2()
```
